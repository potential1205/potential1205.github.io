    public static int binarySearch(int[] a, int target) {
        int l = 0, r = a.length - 1;
        while (l <= r) {
            int m = l + (r - l) / 2;
            if (a[m] == target) return m;
            if (a[m] < target) l = m + 1;
            else r = m - 1;
        }
        return -1;
    }

    // 2) lower_bound: target ì´ìƒì´ ì²˜ìŒ ë‚˜íƒ€ë‚˜ëŠ” ìœ„ì¹˜(ì‚½ì… ìœ„ì¹˜)
    //   ì˜ˆ) [1,2,4,4,5], target=4 -> 2
    public static int lowerBound(int[] a, int target) {
        int l = 0, r = a.length; // [l, r)
        while (l < r) {
            int m = l + (r - l) / 2;
            if (a[m] < target) l = m + 1;
            else r = m;
        }
        return l; // ì‚½ì… ìœ„ì¹˜
    }

    // 3) upper_bound: target ì´ˆê³¼ê°€ ì²˜ìŒ ë‚˜íƒ€ë‚˜ëŠ” ìœ„ì¹˜
    //   ì˜ˆ) [1,2,4,4,5], target=4 -> 4
    public static int upperBound(int[] a, int target) {
        int l = 0, r = a.length; // [l, r)
        while (l < r) {
            int m = l + (r - l) / 2;
            if (a[m] <= target) l = m + 1;
            else r = m;
        }
        return l; // ì‚½ì… ìœ„ì¹˜
    }

    // 4) ì¡´ì¬ ì—¬ë¶€/ê°œìˆ˜ êµ¬í•˜ê¸°: lower/upper í™œìš©
    public static boolean exists(int[] a, int target) {

        return lb < a.length && a[lb] == target;
    }

    public static int countEqual(int[] a, int target) {
        return upperBound(a, target) - lowerBound(a, target);
    }

    }
